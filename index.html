// SPDX-License-Identifier: MIT pragma solidity ^0.8.0; interface IERC20 { function transfer(address recipient, uint256 amount) external returns (bool); function balanceOf(address account) external view returns (uint256); } contract RupAirdrop { address public owner; IERC20 public rupToken; uint256 public airdropAmount = 1000 * 10 ** 18; // 1000 RUP, adjust decimals as per token mapping(address => bool) public claimed; constructor(address _tokenAddress) { owner = msg.sender; rupToken = IERC20(_tokenAddress); } modifier onlyOwner() { require(msg.sender == owner, "Not the owner"); _; } function claim() external { require(!claimed[msg.sender], "Already claimed"); require(rupToken.balanceOf(address(this)) >= airdropAmount, "Not enough tokens in contract"); claimed[msg.sender] = true; rupToken.transfer(msg.sender, airdropAmount); } function updateAmount(uint256 _amount) external onlyOwner { airdropAmount = _amount; } function withdrawTokens() external onlyOwner { rupToken.transfer(owner, rupToken.balanceOf(address(this))); } }
